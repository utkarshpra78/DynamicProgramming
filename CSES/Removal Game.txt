void square(){
    //write code here
        int n;
    cin >> n;
    vector<ll> values(n);
    vector<vector<ll>> maximum(n,vector<ll>(n));
    ll total= 0;
    for (int i = 0; i < n; i++)
    {
        cin >> values[i];
        maximum[i][i] = values[i];
        total+= values[i];
    }
    for (int left = n-1; left >= 0; left--)
    {
        for (int right = left + 1; right < n; right++)
        {
            ll first_element = values[left] - maximum[left + 1][right]; 
            ll last_element = values[right] - maximum[left][right - 1]; 
            maximum[left][right] = max( first_element,
                                                    last_element );
        }
    }
    cout << (total + maximum[0][n - 1]) / 2;    
    
     
 }
